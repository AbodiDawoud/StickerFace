import SwiftUI



struct StickerEffectDemoView: View {
    var body: some View {
        Form {
            Section {
                StickerEffect(effect: .iridescent)
                    .listRowInsets(EdgeInsets())
                    .frame(width: 250, height: 250)
                    .listRowBackground(Color.clear)
                    .frame(maxWidth: .infinity)
            }
        }
    }
    
    func checkImageAnalysis() {
        let frameworkPath = "/System/Library/PrivateFrameworks/VisionKitCore.framework/VisionKitCore"
        let handler = dlopen(frameworkPath, RTLD_NOW)
        
        let VKCStickerEffect = NSClassFromString("VKCStickerEffect") as! NSObject.Type
        let strokeEffect = VKCStickerEffect.value(forKey: "strokeEffect") as! NSObject

        let applySelector = NSSelectorFromString("applyToImage:")
        let image: Data = UIImage(resource: .logo).pngData()!
        let completionHandler: (AnyObject) -> Void = {
            print("Got something back", $0)
        }
        strokeEffect.perform(applySelector, with: image, with: completionHandler)
        print("Request Successful")
    }
}

struct StickerEffect: UIViewRepresentable {
    var effect: StickEffect = .comic
    
    func makeUIView(context: Context) -> UIView {
        let frameworkPath = "/System/Library/PrivateFrameworks/VisionKitCore.framework/VisionKitCore"
        let handler = dlopen(frameworkPath, RTLD_NOW)
        
        let VKCStickerEffect = NSClassFromString("VKCStickerEffect") as! NSObject.Type
        let effect = VKCStickerEffect.value(forKey: effect.className) as! NSObject
        let image = UIImage(resource: .kawaii)
        
        
        
        let controller = NSClassFromString("VKCStickerEffectView") as! UIView.Type
        let instance = controller.init()
        
        instance.setValue(image, forKey: "image")
        instance.setValue(effect, forKey: "effect")
        instance.setValue(false, forKey: "paused")
        
        
        
        return instance
    }
    
    func updateUIView(_ uiView: UIView, context: Context) {}
}

enum StickEffect {
    case comic
    case puffy
    case stroke
    case iridescent
    case none
    
    var className: String {
        switch self {
        case .comic: return "comicEffect"
        case .puffy: return "puffyEffect"
        case .stroke: return "strokeEffect"
        case .iridescent: return "iridescentEffect"
        case .none: return "noneEffect"
        }
    }
}
